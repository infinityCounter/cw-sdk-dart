sortPublicOrders(List<PublicOrder> orders, [bool asc = true]) {
  orders.sort((a, b) {
    if (a.price < b.price) {
      return (asc) ? -1 : 1;
    } else if (a.price > b.price) {
      return (asc) ? 1 : -1;
    } else {
      return 0;
    }
  });
}
